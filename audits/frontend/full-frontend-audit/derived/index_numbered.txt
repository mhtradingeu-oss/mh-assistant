   253	    (function bootstrapControlCenterFallback() {
   254	      function hideLoadingOverlay() {
   255	        var overlay = document.getElementById("loadingOverlay");
   256	        if (!overlay) return;
   257	
   258	        overlay.classList.remove("is-visible");
   259	        overlay.setAttribute("aria-hidden", "true");
   260	        overlay.style.display = "none";
   261	        overlay.style.pointerEvents = "none";
   262	      }
   263	
   264	      function showFatalStartupError(message, details) {
   265	        hideLoadingOverlay();
   266	
   267	        var panel = document.getElementById("fatalErrorPanel");
   268	        var text = document.getElementById("fatalErrorText");
   269	        var diagnostics = document.getElementById("fatalErrorDetails");
   270	        var globalError = document.getElementById("globalError");
   271	
   272	        if (text) {
   273	          text.textContent = message || "The app failed to initialize.";
   274	        }
   275	
   276	        if (diagnostics) {
   277	          diagnostics.textContent = details || "No additional diagnostics available.";
   278	        }
   279	
   280	        if (panel) {
   281	          panel.hidden = false;
   282	          panel.classList.add("is-visible");
   283	          panel.setAttribute("aria-hidden", "false");
   284	        }
   285	
   286	        if (globalError) {
   287	          globalError.textContent = message || "The app failed to initialize.";
   288	          globalError.style.display = "block";
   289	        }
   290	      }
   291	
   292	      var retryBtn = document.getElementById("fatalRetryBtn");
   293	      if (retryBtn) {
   294	        retryBtn.addEventListener("click", function () {
   295	          window.location.reload();
   296	        });
   297	      }
   298	
   299	      var accessKeyBtn = document.getElementById("fatalAccessKeyBtn");
   300	      if (accessKeyBtn) {
   301	        accessKeyBtn.addEventListener("click", function () {
   302	          try {
   303	            var entered = window.prompt("Paste Control Center access key:", "");
   304	            if (entered && entered.trim()) {
   305	              window.localStorage.setItem("mh-control-write-key", entered.trim());
   306	              window.location.reload();
   307	            }
   308	          } catch (_) {
   309	            // Ignore localStorage errors in restricted contexts.
   310	          }
   311	        });
   312	      }
   313	
   314	      window.addEventListener("error", function (event) {
   315	        if (window.__mhControlCenterStarted) return;
   316	
   317	        var target = event && event.target;
   318	        var src = target && target.src ? String(target.src) : "";
   319	        var tagName = target && target.tagName ? String(target.tagName).toUpperCase() : "";
   320	        var isModuleLoadError =
   321	          tagName === "SCRIPT" &&
   322	          src.indexOf("/control-center/app.js") !== -1;
   323	
   324	        if (!isModuleLoadError) return;
   325	
   326	        showFatalStartupError(
   327	          "Control Center JavaScript failed to load.",
   328	          src || "Module script load error"
   329	        );
   330	      }, true);
   331	
   332	      window.addEventListener("unhandledrejection", function (event) {
   333	        if (window.__mhControlCenterStarted) return;
   334	
   335	        var reason = event && event.reason;
   336	        var message = reason && reason.message
   337	          ? reason.message
   338	          : String(reason || "Unhandled startup rejection");
   339	
   340	        showFatalStartupError("Control Center startup promise failed.", message);
   341	      });
   342	    })();

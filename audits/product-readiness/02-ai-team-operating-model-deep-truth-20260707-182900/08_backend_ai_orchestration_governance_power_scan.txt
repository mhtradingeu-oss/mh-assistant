// =====================================================
// MEDIA ORCHESTRATOR BRAIN (GLOBAL SYSTEM CONTROLLER)
// CLOUD-GRADE DECISION ENGINE
// =====================================================

const { buildContentMediaContract } =
  require('../contracts/contentMediaContract');

const queue = require('../queue/facade/mediaQueueFacade');
const eventBus = require('../events/mediaEventBus');

class MediaOrchestratorBrain {

  // -------------------------------
  // MAIN ENTRY POINT
  // -------------------------------
  async execute(input) {

    // 1. BUILD CONTRACT (SOURCE OF TRUTH)
    const contract = buildContentMediaContract(input);

    // 2. DECISION ENGINE
    const decision = this.decide(contract);

    // 3. EMIT START EVENT
    eventBus.emit('MEDIA_ORCHESTRATION_STARTED', {
      contract,
      decision
    });

    // 4. CREATE QUEUE JOB
    const job = await queue.enqueue({
      type: 'media_generation',
      contract,
      decision
    });

    // 5. RETURN CONTROL OBJECT
    return {
      success: true,
      contract,
      decision,
      jobId: job?.id || null
    };
  }

  // -------------------------------
  // INTELLIGENT DECISION ENGINE
  // -------------------------------
  decide(contract) {

    const mediaTypes = contract.media.types;

    let strategy = {
      mode: 'standard',
      provider: 'default',
      priority: 'normal'
    };

    // VIDEO HEAVY CONTENT
    if (mediaTypes.includes('video')) {
      strategy.mode = 'gpu';
      strategy.provider = 'video-engine';
      strategy.priority = 'high';
    }

    // MULTI MEDIA CONTENT
    if (mediaTypes.length > 1) {
      strategy.mode = 'distributed';
      strategy.provider = 'multi-provider';
    }

    // PLATFORM OPTIMIZATION
    if (contract.platform.primary === 'tiktok') {
      strategy.provider = 'short-form-engine';
      strategy.priority = 'high';
    }

    return strategy;
  }

  // -------------------------------
  // HEALTH CHECK
  // -------------------------------
  health() {
    return {
      status: 'active',
      queue: !!queue,
      events: !!eventBus
    };
  }
}

// =====================================================
// MEDIA GOVERNANCE ENGINE (FINAL CONTROL BRAIN)
// CLOUD-GRADE SAFETY + CONTROL LAYER
// =====================================================

class MediaGovernanceEngine {

  constructor() {
    this.limits = {
      maxConcurrentJobs: 5,
      maxVideoJobsPerMinute: 2,
      maxCostLevel: 'high'
    };

    this.activeJobs = new Map();
    this.jobHistory = [];
  }

  // -------------------------------
  // MAIN GATE (BEFORE EXECUTION)
  // -------------------------------
  validate(contract, decision) {

    // 1. VALIDATE CONTRACT
    if (!contract || !contract.media) {
      return this.reject('INVALID_CONTRACT');
    }

    // 2. COST CHECK
    if (decision.mode === 'gpu' && !this.allowGPU()) {
      return this.reject('GPU_LIMIT_EXCEEDED');
    }

    // 3. CONCURRENCY CHECK
    if (this.activeJobs.size >= this.limits.maxConcurrentJobs) {
      return this.reject('CONCURRENCY_LIMIT');
    }

    // 4. VIDEO RATE LIMIT
    if (contract.media.types.includes('video')) {
      const recentVideos = this.getRecentVideoJobs();

      if (recentVideos >= this.limits.maxVideoJobsPerMinute) {
        return this.reject('VIDEO_RATE_LIMIT');
      }
    }

    // 5. DUPLICATION CHECK
    if (this.isDuplicate(contract)) {
      return this.reject('DUPLICATE_JOB');
    }

    return {
      allowed: true,
      priority: this.resolvePriority(contract, decision)
    };
  }

  // -------------------------------
  // GPU CHECK
  // -------------------------------
  allowGPU() {
    return true; // placeholder for real infra check
  }

  // -------------------------------
  // DUPLICATION CHECK
  // -------------------------------
  isDuplicate(contract) {
    return this.jobHistory.some(job =>
      job.hash === JSON.stringify(contract)
    );
  }

  // -------------------------------
  // PRIORITY ENGINE
  // -------------------------------
  resolvePriority(contract, decision) {

    if (contract.platform?.primary === 'tiktok') {
      return 'high';
    }

    if (decision.mode === 'gpu') {
      return 'high';
    }

    return 'normal';
  }

  // -------------------------------
  // TRACK JOB
  // -------------------------------
  track(jobId, contract) {
    this.activeJobs.set(jobId, contract);
    this.jobHistory.push({
      id: jobId,
      hash: JSON.stringify(contract),
      timestamp: Date.now()
    });
  }

  // -------------------------------
  // RELEASE JOB
  // -------------------------------
  release(jobId) {
    this.activeJobs.delete(jobId);
  }

  // -------------------------------
  // REJECT HANDLER
  // -------------------------------
  reject(reason) {
    return {
      allowed: false,
      reason
    };
  }

  // -------------------------------
  // METRICS
  // -------------------------------
  status() {
    return {
      activeJobs: this.activeJobs.size,
      history: this.jobHistory.length
    };
  }
}

    tiktok: { aspect: "9:16", style: "fast_hook" },
    youtube: { aspect: "16:9", style: "story" },

  };

  if (input.need_video) types.push("video");

  if (types.length === 0) types.push("image");

// ===============================
// WORKER QUEUE SYSTEM (PHASE 5)
// ===============================

const { EventEmitter } = require('events');
const { mediaEventBus, EVENTS } = require('../events/mediaEventBus');

class MediaQueue extends EventEmitter {
  constructor() {
    super();
    this.jobs = [];
    this.processing = false;
  }

  // add job to queue
  add(job) {
    this.jobs.push(job);
    this.process();
  }

  // process queue sequentially
  async process() {
    if (this.processing) return;
    this.processing = true;

    while (this.jobs.length > 0) {
      const job = this.jobs.shift();
      await this.execute(job);
    }

    this.processing = false;
  }

  // worker execution layer
  async execute(job) {
    try {
      mediaEventBus.emitEvent(EVENTS.PUBLISH_STARTED, job);

      job.status = 'processing';
      job.progress = 10;

      // simulate async worker execution
      await this.simulateWork(job);

      job.status = 'completed';
      job.progress = 100;

      mediaEventBus.emitEvent(EVENTS.PUBLISH_COMPLETED, job);

    } catch (err) {
      job.status = 'failed';
      job.error = err.message;

      mediaEventBus.emitEvent(EVENTS.PUBLISH_FAILED, job);
    }
  }

  // simulate real worker load
  simulateWork(job) {
    return new Promise(resolve => {
      setTimeout(() => {
        job.progress = 70;
        resolve();
      }, 800);
    });
  }
}

const mediaQueue = new MediaQueue();

module.exports = {
  mediaQueue
};

const EventEmitter = require('events');

class MediaEventBus extends EventEmitter {
  constructor() {
    super();
    this.setMaxListeners(100);
  }

  emitEvent(type, payload = {}) {
    const event = {
      type,
      payload,
      timestamp: new Date().toISOString()
    };

    this.emit(type, event);
    this.emit('*', event); // global listener

    return event;
  }

  onEvent(type, handler) {
    this.on(type, handler);
  }
}

    44	/* =========================================
    45	   Runtime Snapshot Helpers
    46	========================================= */
    47	
    48	function getCommandRuntimeSnapshot({
    49	  commandBar = null,
    50	  commandBackdrop = null,
    51	  aiDock = null,
    52	  aiDockToggle = null,
    53	  aiDockPanel = null
    54	} = {}) {
    55	  return {
    56	    version: COMMAND_RUNTIME_VERSION,
    57	
    58	    commandBar: {
    59	      exists: Boolean(commandBar),
    60	      hidden: Boolean(commandBar?.hidden),
    61	      ariaHidden: String(commandBar?.getAttribute?.("aria-hidden") || ""),
    62	      className: String(commandBar?.className || "")
    63	    },
    64	
    65	    commandBackdrop: {
    66	      exists: Boolean(commandBackdrop),
    67	      hidden: Boolean(commandBackdrop?.hidden),
    68	      ariaHidden: String(commandBackdrop?.getAttribute?.("aria-hidden") || ""),
    69	      className: String(commandBackdrop?.className || "")
    70	    },
    71	
    72	    aiDock: {
    73	      exists: Boolean(aiDock),
    74	      open: Boolean(aiDock?.classList?.contains("is-open")),
    75	      dataOpen: String(aiDock?.getAttribute?.("data-open") || ""),
    76	      className: String(aiDock?.className || ""),
    77	      toggle: {
    78	        exists: Boolean(aiDockToggle),
    79	        ariaExpanded: String(aiDockToggle?.getAttribute?.("aria-expanded") || "")
    80	      },
    81	      panel: {
    82	        exists: Boolean(aiDockPanel),
    83	        hidden: Boolean(aiDockPanel?.hidden),
    84	        inert: Boolean(aiDockPanel?.inert),
    85	        ariaHidden: String(aiDockPanel?.getAttribute?.("aria-hidden") || ""),
    86	        pointerEvents: String(aiDockPanel?.style?.pointerEvents || "")
    87	      }
    88	    }
    89	  };
    90	}

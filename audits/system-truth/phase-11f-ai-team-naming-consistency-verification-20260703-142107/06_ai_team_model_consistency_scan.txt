export function getAiTeamRole(roleId) {
  return AI_TEAM_ROLES.find((role) => role.id === roleId) || null;
}

export function getAiRolesForPage(pageId) {
  const ids = AI_PAGE_SPECIALIST_MAP[pageId] || [];
  return ids.map(getAiTeamRole).filter(Boolean);
}

export function getPrimaryAiRoleForPage(pageId) {
  return getAiRolesForPage(pageId)[0] || AI_TEAM_ROLES[0];
}
